<template>
    <div class="pr-[10px] pl-[10px]" 
     v-if="permission.can(tassiliInput.tassiliFormList[formKey]['info']['permissions'])">

      <WizardStep :formKey="formKey" />

      <form @submit.prevent="submit('creer')" @keydown.enter.prevent>
       <div class="conteneurForm">
        <template
         v-for="([key, field], index) in Object.entries(tassiliInput.form[formKey])
             .filter(([k, f]) =>
              f.type !== 'Hidden' &&
            (tassiliInput.tassiliFormList[formKey]['info']['wizard']['wizardForm'][tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent']] ?? []).includes(k)
               )" :key="key">
       
       <InputText v-if="field.type === 'Text'" 
       :FormKey="formKey"  :cle="key" />

       <Textarea v-if="field.type === 'Textarea'" 
       :FormKey="formKey" :cle="key" />

        <InputNumber v-if="field.type === 'Number'" 
       :FormKey="formKey" :cle="key" />

       <Password v-if="field.type === 'Password'" 
       :FormKey="formKey" :cle="key" />

       <InputSelect v-if="field.type === 'Select'" 
       :FormKey="formKey" :cle="key" />

       <InputDate v-if="field.type === 'Date'" 
       :FormKey="formKey" :cle="key" />

       <InputRadio v-if="field.type === 'Radio'" 
       :FormKey="formKey" :cle="key" />

       <Checkbox v-if="field.type === 'Checkbox'" 
       :FormKey="formKey" :cle="key" />

       <CheckboxList v-if="field.type === 'CheckboxList'" 
       :FormKey="formKey" :cle="key" />

       <RichEditor v-if="field.type === 'Quill'" 
       :FormKey="formKey" :cle="key" />

       <InputFile v-if="field.type === 'File'" 
       :FormKey="formKey" :cle="key" />
      
       <Repeteur v-if="field.type === 'Repeater'" 
       :FormKey="formKey" :cle="key" />

       <Signature v-if="field.type === 'Signature'" 
       :FormKey="formKey" :cle="key" />

       <MultipleFile v-if="field.type === 'MultipleFile'" 
       :FormKey="formKey" :cle="key" />

      </template>
      </div>

       <LoadAction :formKey="formKey" />

       <div class="pt-[10px] pt-[0px] flex flex-wrap gap-[5px]"
        v-if="doubleClick"> 
           <div v-if="(tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent'] != 1)" @click="reculer()" >  
            <button  class="bg-[black] min-w-[100px] text-white p-[9px] rounded-[2px]" type="button">
              Previous
            </button>
          </div>
           <div v-if="(tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent'] != tassiliInput.tassiliFormList[formKey]['info']['wizard'].wizardCount)">
            <button class="bg-[black] min-w-[100px] text-white p-[9px] rounded-[2px]" type="button"
              @click="submit('next')">
              Next
            </button>
          </div>
          <div v-if="(tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent'] == tassiliInput.tassiliFormList[formKey]['info']['wizard'].wizardCount) ||
                tassiliInput.tassiliFormList[formKey]['info']['wizard'].wizardStop.includes(tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent'])">
          <button class="bg-[blue] min-w-[100px] text-white p-[9px] rounded-[2px]" type="submit">
              {{tassiliInput.tassiliFormList[formKey]['info']['createLabel']  }}
            </button>
          </div>
          <div  v-if="(tassiliInput.tassiliFormList[formKey]['info']['createOther'] == 'yes'
             && tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent'] == tassiliInput.tassiliFormList[formKey]['info']['wizard'].wizardCount) ||
             (tassiliInput.tassiliFormList[formKey]['info']['createOther'] == 'yes'
             && tassiliInput.tassiliFormList[formKey]['info']['wizard'].wizardStop.includes(tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent']))">
             <button type="button" @click="submit('other')" class="border-[1px] border-black min-w-[100px] text-black p-[9px] rounded-[2px]">
              {{tassiliInput.tassiliFormList[formKey]['info']['createOtherLabel']  }} 
            </button>
          </div>
         

          </div>

      </form>


    </div>
</template>

<script setup>
import { TassiliInput } from '../stores/tassiliInput';
import { TassiliRoutes } from '@/vendor/TassiliLibs/stores/tassiliRoutes';
import InputText from '../Inputs/InputText.vue';
import Textarea from '../Inputs/Textarea.vue';
import InputNumber from '../Inputs/InputNumber.vue';
import Password from '../Inputs/Password.vue';
import InputSelect from '../Inputs/InputSelect.vue';
import InputDate from '../Inputs/InputDate.vue';
import InputRadio from '../Inputs/InputRadio.vue';
import Checkbox from '../Inputs/Checkbox.vue';
import CheckboxList from '../Inputs/CheckboxList.vue';
import RichEditor from '../Inputs/RichEditor.vue';
import InputFile from '../Inputs/InputFile.vue';
import MultipleFile from '../Inputs/MultipleFile.vue';
import Signature from '../Inputs/Signature.vue';
import Repeteur from '../Inputs/Repeteur.vue';
import { router, usePage } from '@inertiajs/vue3';
import { Notyf } from 'notyf';
import 'notyf/notyf.min.css';
import { computed ,ref} from "vue";
import WizardStep from './WizardStep.vue';
import LoadAction from './LoadAction.vue';
import { Permission } from '../stores/permission';

const props = defineProps({
  formKey : {
    type: String,
    required: true
  }
})

const permission = Permission()
const tassiliInput = TassiliInput()
const page = usePage()
const formKey = props.formKey

tassiliInput.form[formKey] = JSON.parse(JSON.stringify(tassiliInput.tassiliFormList[formKey]['fields']))

const smCols = computed(() =>
 tassiliInput.tassiliFormList?.[formKey]?.info?.grid?.sm || 1
);
const mdCols = computed(() =>
 tassiliInput.tassiliFormList?.[formKey]?.info?.grid?.md || 1
);
const lgCols = computed(() =>
 tassiliInput.tassiliFormList?.[formKey]?.info?.grid?.lg || 1
);
const xlCols = computed(() =>
 tassiliInput.tassiliFormList?.[formKey]?.info?.grid?.xl || 1
);

function reculer() {
  tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent'] -= 1 
  tassiliInput.tassiliFormList[formKey]['info']['isAnimated'] = 'off'
}

function nextValidate() {
  tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent'] += 1 
  tassiliInput.tassiliFormList[formKey]['info']['isAnimated'] = 'off'
}

function cleanQuillContent(html) {
  if (typeof html !== 'string') return html;
  return html.replace(/<p>\s*<\/p>/g, '').replace(/<p><br><\/p>/g, '').trim();
}

const doubleClick = ref(true)

function insert(action) {
 if (doubleClick.value !== true) {
     return;
  }
  doubleClick.value = false
  const formData = new FormData();

  tassiliInput.tassiliFormList[formKey]['info']['isAnimated'] = 'on'

  if (action == 'next') {
    formData.append('tassiliSaveActive', 'no');
  }
  if (action != 'next') {
    formData.append('tassiliSaveActive', 'yes');
  }
  
  formData.append('tassiliWizardStep',tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent']);

  formData.append(
    'urlValidationurlValidationurlValidationTassili17485RRY4R4RD9448RK48K4RFRFIRU',
    formKey
  );
  formData.append(
    'permissions17485RRY4R4RD9448RK48K4RFRFIRU',
    JSON.stringify(tassiliInput.tassiliFormList[formKey]['info']['permissions']) 
  );
  Object.keys(tassiliInput.form[formKey]).forEach((key) => {

    const tab1 = ['MultipleFile']
    if (tab1.includes(tassiliInput.form[formKey][key]['type'])) {
    if(!tassiliInput.form[formKey][key]['value'] || tassiliInput.form[formKey][key]['value'].length === 0) {
      formData.append(key, '');
    }
    else if (Array.isArray(tassiliInput.form[formKey][key]['value'])) {
      tassiliInput.form[formKey][key]['value'].forEach((file, index) => {
        formData.append(`${key}[]`, file);
      });
    }
 
    
    } 


    const tab2 = ['File'];
    if(tab2.includes(tassiliInput.form[formKey][key]['type'])) {
      formData.append(key , tassiliInput.form[formKey][key]['value']);
    }


    const tab3 = [
      'Text','Date','Hidden','Select','Number','Radio','Checkbox','CheckboxList','Password','Textarea'
    ,'Signature'];
    if (tab3.includes(tassiliInput.form[formKey][key]['type'])) {
      formData.append(key, tassiliInput.form[formKey][key]['value']);
    }


    const tab4 = ['Quill'];
    if(tab4.includes(tassiliInput.form[formKey][key]['type'])) {
      formData.append(key, cleanQuillContent(tassiliInput.form[formKey][key]['value'] || ''));
    }



if (tassiliInput.form[formKey][key]['type'] === 'Repeater' ) {

   tassiliInput.form[formKey][key]['value'].forEach((item, i) => {
    Object.entries(item).forEach(([subKey, subValue]) => {

       if(tassiliInput.form[formKey][key]['fields'][subKey]['type'] === 'Quill') {
          subValue  = cleanQuillContent(subValue || '')
      }

      formData.append(`${key}[${i}][${subKey}]`, subValue);

    });
  });


}



  });

  router.post('/tassili17485RRY4R4RD9448RK48K4RFRFIRU/valideur/framework', formData, {
    forceFormData: true,
    onError: (errors) => {
      tassiliInput.setError(formKey ,errors);
      tassiliInput.tassiliFormList[formKey]['info']['isAnimated'] = 'off'
    },
    onSuccess: () => {
      
       if (action === 'creer') {
          tassiliInput.cancelError(formKey);
          tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent'] = 1
          tassiliInput.form[formKey] = JSON.parse(JSON.stringify(tassiliInput.tassiliFormList[formKey]['fields']))
          const notyf = new Notyf({ position: { x: 'right', y: 'top' } });
          notyf.success(tassiliInput.tassiliFormList[formKey]['info']['createMessage']);
          tassiliInput.tassiliFormList[formKey]['info']['isAnimated'] = 'off'
      
          if(tassiliInput.tassiliFormList[formKey]['info']['routeSession'] != '') {
             const tassiliroutes = TassiliRoutes();
             tassiliroutes.visit(tassiliInput.tassiliFormList[formKey]['info']['routeSessionData']['model'],
             tassiliInput.tassiliFormList[formKey]['info']['routeSessionData']['url']);
             return;
          }
        
          if(tassiliInput.tassiliFormList[formKey]['info']['route'] != '') {
             router.visit(tassiliInput.tassiliFormList[formKey]['info']['route'])
             return;
          }

      } else if (action === 'other') {
          tassiliInput.cancelError(formKey);
          tassiliInput.tassiliFormList[formKey]['info']['wizardCurrent'] = 1
          tassiliInput.form[formKey] = JSON.parse(JSON.stringify(tassiliInput.tassiliFormList[formKey]['fields']))
          const notyf = new Notyf({ position: { x: 'right', y: 'top' } });
          notyf.success(tassiliInput.tassiliFormList[formKey]['info']['createMessage']);
          tassiliInput.tassiliFormList[formKey]['info']['isAnimated'] = 'off'
      }
     else if (action === 'next') {
        nextValidate();
      }
    },
      onFinish: () => {
      doubleClick.value = true
    },
  });
}


function submit(action) {
   insert(action);
  }


</script>
